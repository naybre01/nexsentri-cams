# Build Stage
FROM node:18-alpine as builder

WORKDIR /app

# Copy dependency definitions
COPY package.json ./
# Copy lock file if it exists, ignore if not
COPY package-lock.json* ./

# Install dependencies
RUN npm install

# Copy source code
COPY . .

# Build the application
RUN npm run build

# Production Stage
FROM nginx:alpine

# Copy built assets from builder stage
COPY --from=builder /app/dist /usr/share/nginx/html

# Copy Nginx configuration
# NOTE: Ensure this filename matches what you have on disk (nginx.conf.txt or nginx.conf)
COPY nginx.conf.txt /etc/nginx/conf.d/default.conf

# Expose port 80
EXPOSE 80

# Start Nginx
# We use sed to replace __API_KEY__ in index.html with the actual environment variable at runtime
CMD ["/bin/sh", "-c", "sed -i \"s|__API_KEY__|$API_KEY|g\" /usr/share/nginx/html/index.html && nginx -g 'daemon off;'"]
